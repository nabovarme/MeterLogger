120 //Parses a label into a C-string containing a dotted
121 //Returns pointer to start of next fields in packet
122 static char* ICACHE_FLASH_ATTR labelToStr(char *packet, char *labelPtr, int packetSz, char *res, int resMaxLen) {
123  int i, j, k;
124  char *endPtr=NULL;
125  i=0;
126  do {
127  if ((*labelPtr&0xC0)==0) {
128  j=*labelPtr++; //skip past length
129  //Add separator period if there already is data in res
130  if (i<resMaxLen && i!=0) res[i++]='.';
131  //Copy label to res
132  for (k=0; k<j; k++) {
133  if ((labelPtr-packet)>packetSz) return NULL;
134  if (i<resMaxLen) res[i++]=*labelPtr++;
135  }
136  } else if ((*labelPtr&0xC0)==0xC0) {
137  //Compressed label pointer
138  endPtr=labelPtr+2;
139  int offset=my_ntohs(((uint16_t *)labelPtr))&0x3FFF;
140  //Check if offset points to somewhere outside of the packet
141  if (offset>packetSz) return NULL;
142  labelPtr=&packet[offset];
143  }
144  //check for out-of-bound-ness
145  if ((labelPtr-packet)>packetSz) return NULL;
146  } while (*labelPtr!=0);
147  res[i]=0; //zero-terminate
148  if (endPtr==NULL) endPtr=labelPtr+1;
149  return endPtr;
150 }

153 //Converts a dotted hostname to the weird label form dns uses.
154 static char ICACHE_FLASH_ATTR *strToLabel(char *str, char *label, int maxLen) {
155  char *len=label; //ptr to len byte
156  char *p=label+1; //ptr to next label byte to be written
157  while (1) {
158  if (*str=='.' || *str==0) {
159  *len=((p-len)-1); //write len of label bit
160  len=p; //pos of len for next part
161  p++; //data ptr is one past len
162  if (*str==0) break; //done
163  str++;
164  } else {
165  *p++=*str++; //copy byte
166 // if ((p-label)>maxLen) return NULL; //check out of bounds
167  }
168  }
169  *len=0;
170  return p; //ptr to first free byte in resp
171 }

175 #ifndef FREERTOS
176 static void ICACHE_FLASH_ATTR captdnsRecv(void* arg, char *pusrdata, unsigned short length) {
177  struct espconn *conn=(struct espconn *)arg;
178 #else
179 static void ICACHE_FLASH_ATTR captdnsRecv(struct sockaddr_in *premote_addr, char *pusrdata, unsigned short length) {
180 #endif
181  char buff[DNS_LEN];
182  char reply[DNS_LEN];
183  int i;
184  char *rend=&reply[length];
185  char *p=pusrdata;
186  DnsHeader *hdr=(DnsHeader*)p;
187  DnsHeader *rhdr=(DnsHeader*)&reply[0];
188  p+=sizeof(DnsHeader);
189 // httpd_printf("DNS packet: id 0x%X flags 0x%X rcode 0x%X qcnt %d ancnt %d nscount %d arcount %d len %d\n",
190 // my_ntohs(&hdr->id), hdr->flags, hdr->rcode, my_ntohs(&hdr->qdcount), my_ntohs(&hdr->ancount), my_ntohs(&hdr->nscount), my_ntohs(&hdr->arcount), length);
191  //Some sanity checks:
192  if (length>DNS_LEN) return; //Packet is longer than DNS implementation allows
193  if (length<sizeof(DnsHeader)) return; //Packet is too short
194  if (hdr->ancount || hdr->nscount || hdr->arcount) return; //this is a reply, don't know what to do with it
195  if (hdr->flags&FLAG_TC) return; //truncated, can't use this
196  //Reply is basically the request plus the needed data
197  memcpy(reply, pusrdata, length);
198  rhdr->flags|=FLAG_QR;
199  for (i=0; i<my_ntohs(&hdr->qdcount); i++) {
200  //Grab the labels in the q string
201  p=labelToStr(pusrdata, p, length, buff, sizeof(buff));
202  if (p==NULL) return;
203  DnsQuestionFooter *qf=(DnsQuestionFooter*)p;
204  p+=sizeof(DnsQuestionFooter);
205 // httpd_printf("DNS: Q (type 0x%X class 0x%X) for %s\n", my_ntohs(&qf->type), my_ntohs(&qf->class), buff);
206  if (my_ntohs(&qf->type)==QTYPE_A) {
207  //They want to know the IPv4 address of something.
208  //Build the response.
209  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
210  if (rend==NULL) return;
211  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
212  rend+=sizeof(DnsResourceFooter);
213  setn16(&rf->type, QTYPE_A);
214  setn16(&rf->class, QCLASS_IN);
215  setn32(&rf->ttl, 0);
216  setn16(&rf->rdlength, 4); //IPv4 addr is 4 bytes;
217  //Grab the current IP of the softap interface
218  struct ip_info info;
219  wifi_get_ip_info(SOFTAP_IF, &info);
220  *rend++=ip4_addr1(&info.ip);
221  *rend++=ip4_addr2(&info.ip);
222  *rend++=ip4_addr3(&info.ip);
223  *rend++=ip4_addr4(&info.ip);
224  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
225 // httpd_printf("Added A rec to resp. Resp len is %d\n", (rend-reply));
226  } else if (my_ntohs(&qf->type)==QTYPE_NS) {
227  //Give ns server. Basically can be whatever we want because it'll get resolved to our IP later anyway.
228  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
229  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
230  rend+=sizeof(DnsResourceFooter);
231  setn16(&rf->type, QTYPE_NS);
232  setn16(&rf->class, QCLASS_IN);
233  setn16(&rf->ttl, 0);
234  setn16(&rf->rdlength, 4);
235  *rend++=2;
236  *rend++='n';
237  *rend++='s';
238  *rend++=0;
239  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
240 // httpd_printf("Added NS rec to resp. Resp len is %d\n", (rend-reply));
241  } else if (my_ntohs(&qf->type)==QTYPE_URI) {
242  //Give uri to us
243  rend=strToLabel(buff, rend, sizeof(reply)-(rend-reply)); //Add the label
244  DnsResourceFooter *rf=(DnsResourceFooter *)rend;
245  rend+=sizeof(DnsResourceFooter);
246  DnsUriHdr *uh=(DnsUriHdr *)rend;
247  rend+=sizeof(DnsUriHdr);
248  setn16(&rf->type, QTYPE_URI);
249  setn16(&rf->class, QCLASS_URI);
250  setn16(&rf->ttl, 0);
251  setn16(&rf->rdlength, 4+16);
252  setn16(&uh->prio, 10);
253  setn16(&uh->weight, 1);
254  memcpy(rend, "http://esp.nonet", 16);
255  rend+=16;
256  setn16(&rhdr->ancount, my_ntohs(&rhdr->ancount)+1);
257 // httpd_printf("Added NS rec to resp. Resp len is %d\n", (rend-reply));
258  }
259  }
260  //Send the response
261 #ifndef FREERTOS
262  remot_info *remInfo=NULL;
263  //Send data to port/ip it came from, not to the ip/port we listen on.
264  if (espconn_get_connection_info(conn, &remInfo, 0)==ESPCONN_OK) {
265  conn->proto.udp->remote_port=remInfo->remote_port;
266  memcpy(conn->proto.udp->remote_ip, remInfo->remote_ip, sizeof(remInfo->remote_ip));
267  }
268  espconn_sendto(conn, (uint8*)reply, rend-reply);
269 #else
270  sendto(sockFd,(uint8*)reply, rend-reply, 0, (struct sockaddr *)premote_addr, sizeof(struct sockaddr_in));
271 #endif
272 }

288  server_addr.sin_len = sizeof(server_addr);
289 
290  do {
291  sockFd=socket(AF_INET, SOCK_DGRAM, 0);
292  if (sockFd==-1) {
293 // httpd_printf("captdns_task failed to create sock!\n");
294  vTaskDelay(1000/portTICK_RATE_MS);
295  }
296  } while (sockFd==-1);
297 
298  do {
299  ret=bind(sockFd, (struct sockaddr *)&server_addr, sizeof(server_addr));
300  if (ret!=0) {
301 // httpd_printf("captdns_task failed to bind sock!\n");
302  vTaskDelay(1000/portTICK_RATE_MS);
303  }
304  } while (ret!=0);
305 
306 // httpd_printf("CaptDNS inited.\n");
307  while(1) {
308  memset(&from, 0, sizeof(from));
309  fromlen=sizeof(struct sockaddr_in);
310  ret=recvfrom(sockFd, (u8 *)udp_msg, DNS_LEN, 0,(struct sockaddr *)&from,(socklen_t *)&fromlen);
311  if (ret>0) captdnsRecv(&from,udp_msg,ret);
312  }
313
